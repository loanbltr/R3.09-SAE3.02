"""
Cluster de calcul (côté serveur esclave)

Cette application met en œuvre un serveur esclave utilisant des sockets pour la communication avec un serveur maître.
Le serveur escalve permet de recevoir des fichiers de code source, de les traiter et de renvoyer les résultats au serveur maître.
Le serveur esclave peut également envoyer des informations sur l'utilisation du CPU, de la RAM, le nombre de codes en cours d'exécution, disponibilité, les langages de développement accepté au serveur maître

Classes:
    - ServerSlave: classe chargée de gérer la communication avec le serveur maître et le traitement des fichiers de code source.

Utilisation :
    1. Ouvir un terminal.
    2. Exécuter le script avec la commande "python ServerSlave.py" et les différentes options.
    3. Le serveur esclave se connectera au serveur maître et attendra des fichiers de code source à traiter.

Author:
    Loan BELTRAN

GitHub:
    github.com/loanbltr

Date:
    31/12/2024
"""

class ServerSlave:
    """
        Représente un serveur esclave capable de recevoir des fichiers de code source, de les traiter et de renvoyer les résultats au serveur maître.
    """
    def __init__(self, ipMaster, langages, port, portCpu, maxCpu=100, maxRAM=100, nbCodesMax=99):
        """
        Attributs:
            - ipMaster (str): adresse IP du serveur maître.
            - langages (str): langages de développement accepté par le serveur esclave
            - port (int): port sur lequel le serveur esclave va se connecter.
            - portCpu (int): port sur lequel le serveur esclave va se connecter pour la partie CPU et RAM.
            - maxCpu (int): pourcentage maximum d'utilisation du CPU.
            - maxRam (int): pourcentage maximum d'utilisation de la RAM.
            - nbCodesMax (int): nombre de codes maximum autorisé en même temps.
            - nbCodesNow (int): nombre de codes en cours d'exécution.
            - socketMaster (socket): socket de communication avec le serveur maître.
            - socketMasterCpu (socket): socket de communication avec le serveur maître pour l'utilisation du CPU et de la RAM.
            - disponible (bool): disponibilité du serveur escalve.
            - id (str): identifiant unique du serveur esclave.
        """
        self.ipMaster = ipMaster
        self.maxCpu = maxCpu
        self.maxRam = maxRAM
        self.nbCodesMax = nbCodesMax
        self.nbCodesNow = 0
        self.langages = langages
        self.port = port
        self.portCpu = portCpu

        # Initialisation des sockets de communication avec le serveur maître
        self.socketMaster = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.socketMaster.connect((self.ipMaster, int(self.port)))

        self.socketMasterCpu = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.socketMasterCpu.connect((self.ipMaster, int(self.portCpu)))

        # Envoi de l'identifiant unique du serveur esclave au serveur maître
        self.disponible = True
        self.id = str(random.randint(0, 10000))
        try:
            data = f"{self.id}�{self.langages}"
            self.socketMaster.send(data.encode("utf-8"))
        except Exception as e:
            print(f"Error sending ID to master: {e}")

        # Affichage des informations du serveur esclave
        print("=== Server Slave ===")
        print(f"IP du master: {self.ipMaster}")
        print(f"CPU maximum: {self.maxCpu}")
        print(f"RAM maximum: {self.maxRam}")
        print(f"Nombre de codes maximaux: {self.nbCodesMax}")
        print(f"Langages de développement accepté: {self.langages}")
        print("=====================")

    def start(self):
        """
        Démarre le serveur esclave en écoutant les messages du serveur maître et en envoyant les informations sur l'utilisation du CPU et de la RAM.
        """
        threading.Thread(target=self.receiveMaster).start()
        threading.Thread(target=self.stateCpu).start()

    def receiveMaster(self):
        """
        Reçoit les messages du serveur maître et les traitent.
        """
        try:
            while True:
                data = self.socketMaster.recv(1024).decode("utf-8")
                print(f"Received data from master: {data}")
                if data == None:
                    break
                if data == 'stop':
                    self.stop()
                    break
                else:
                    threading.Thread(target=self.traitementMessage, args=(data,)).start()
        except Exception as e:
            print(f"Error receiving data from master: {e}")

    def sendMaster(self, data):
        """
        Envoie des données au serveur maître.
        """
        if data == 'stateSlave':
            while True:
                try:
                    info = f'stateSlave|{self.disponible}'
                    self.socketMaster.send(info.encode("utf-8"))
                    print(f"Message sent to master: {info}")
                    time.sleep(5)
                except Exception as e:
                    print(f"Error sending data to master: {e}")
                    break
        else:
            try:
                self.socketMaster.send(data.encode("utf-8"))
                print(f"Message sent to master: {data}")
            except Exception as e:
                print(f"Error sending data to master: {e}")

    def stateCpu(self):
        """
        Envoie les informations sur l'utilisation du CPU et de la RAM au serveur maître.
        """
        while True:
            try:
                cpu_usage = psutil.cpu_percent(interval=1)
                ram_usage = psutil.virtual_memory().percent
                if cpu_usage > self.maxCpu or self.nbCodesNow > self.nbCodesMax:
                    self.disponible = False
                else:
                    self.disponible = True
                message = f"{self.id}�{self.disponible}�{str(cpu_usage)}�{str(ram_usage)}"
                self.socketMasterCpu.send(message.encode("utf-8"))

                time.sleep(5)
            except Exception as e:
                print(f"Error sending CPU and RAM usage to master: {e}")

    def traitementMessage(self, data):
        """
        Traite les messages reçus du serveur maître.
        """
        try:
            if data.split('|')[1].split('.')[1] == "py":
                self.traitementPython(data)
            elif data.split('|')[1].split('.')[1] == "java":
                self.traitementJava(data)
            elif data.split('|')[1].split('.')[1] == "c":
                self.traitementC(data)
        except Exception as e:
            print(f"Error processing message: {e}")

    def traitementJava(self, data):
        """
        Traite un fichier de code source Java.
        """
        self.nbCodesNow += 1
        nameTmp = data.split('|')[1].split('.')[0] + str(random.randint(0, 10000)) + ".java"
        if not os.path.exists(f'fileTemp'):
            os.makedirs(f'fileTemp')

        try:
            with open(f'fileTemp/{nameTmp}', "w", encoding='utf-8') as f:
                f.write(data.split('|')[2])
            with open(f"fileTemp/{nameTmp}", "r") as fichier:
                lignes = fichier.readlines()

            for i in range(len(lignes)):
                if lignes[i].startswith("public class"):
                    mots = lignes[i].split(' ')
                    mots[2] = nameTmp.split('.')[0]
                    lignes[i] = " ".join(mots) + "\n"

            with open(f"fileTemp/{nameTmp}", "w") as fichier:
                fichier.writelines(lignes)
        except:
            print("Error during writing file")
            self.nbCodesNow -= 1
        
        result = subprocess.run(["java", f'fileTemp/{nameTmp}'], capture_output=True, text=True)
        if result.returncode == 0:
            self.sendMaster(f"{data.split('�')[0]}�result|{result.stdout}")
        else:
            self.sendMaster(f"{data.split('�')[0]}�result|{result.stderr}")
        os.remove(f'fileTemp/{nameTmp}')
        self.nbCodesNow -= 1

    def traitementPython(self, data):
        """
        Traite un fichier de code source Python.
        """
        self.nbCodesNow += 1
        nameTmp = data.split('|')[1].split('.')[0] + str(random.randint(0, 1000)) + ".py"
        if not os.path.exists(f'fileTemp'):
            os.makedirs(f'fileTemp')
            os.chmod(f'fileTemp', 0o777)
        try:
            with open(f'fileTemp/{nameTmp}', "w", encoding='utf-8') as f:
                f.write(data.split('|')[2])
            os.chmod(f'fileTemp/{nameTmp}', 0o777)
        except Exception as write_error:
            print("Error during writing file:", write_error)
        try:
            result = subprocess.run(['python3', f'fileTemp/{nameTmp}'], capture_output=True, text=True)
        except Exception as e:
            self.sendMaster(str(e))
            self.nbCodesNow -= 1
        else:
            if result.returncode == 0:
                self.sendMaster(f"{data.split('�')[0]}�result|{result.stdout}")
            else:
                self.sendMaster(f"{data.split('�')[0]}�result|{result.stderr}")
        os.remove(f'fileTemp/{nameTmp}')
        self.nbCodesNow -= 1

    def traitementC(self, data):
        """
        Traite un fichier de code source C.
        """
        self.nbCodesNow += 1
        nameTmp = data.split('|')[1].split('.')[0] + str(random.randint(0, 1000)) + ".c"
        try:
            with open(f'fileTemp/{nameTmp}', "w", encoding='utf-8') as f:
                f.write(data.split('|')[2])
        except:
            print("Error during writing file")
        try:
            result = subprocess.run(['gcc', f'fileTemp/{nameTmp}', '-o', f'fileTemp/{nameTmp}.out'], capture_output=True, text=True)
            if result.returncode == 0:
                result = subprocess.run([f'fileTemp/{nameTmp}.out'], capture_output=True, text=True)
            if result.returncode == 0:
                self.sendMaster(f"{data.split('�')[0]}�result|{result.stdout}")
            else:
                self.sendMaster(f"{data.split('�')[0]}�result|{result.stderr}")
        except Exception as e:
            self.sendMaster(str(e))
            self.nbCodesNow -= 1
        os.remove(f'fileTemp/{nameTmp}')
        os.remove(f'fileTemp/{nameTmp}.out')
        self.nbCodesNow -= 1
